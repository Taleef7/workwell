```ts
test("synthetic bundle resources carry QI-Core profiles + required elements", () => {
  // Build a bundle with an enrolled employee + a recent procedure event (reuse the file's
  // existing fixture helpers; the assertions below are the point).
  const bundle = buildSyntheticBundle(/* enrolled employee */, /* config w/ a procedure event */, "2026-06-12");
  const QICORE = "http://hl7.org/fhir/us/qicore/StructureDefinition/";
  const byType = (t: string) => bundle.entry.map((e) => e.resource as Record<string, unknown>).filter((r) => r.resourceType === t);
  const profileOf = (r: Record<string, unknown>) => ((r.meta as { profile?: string[] }) ?? {}).profile?.[0];

  const patient = byType("Patient")[0]!;
  assert.equal(profileOf(patient), `${QICORE}qicore-patient`);
  assert.ok(patient.id && patient.name, "Patient has id + name");

  const cond = byType("Condition")[0]!;
  assert.equal(profileOf(cond), `${QICORE}qicore-condition`);
  assert.ok(cond.clinicalStatus && cond.verificationStatus && cond.code && cond.subject, "Condition required elements");

  const proc = byType("Procedure")[0]!;
  assert.equal(profileOf(proc), `${QICORE}qicore-procedure`);
  assert.ok(proc.status && proc.code && proc.subject, "Procedure required elements");

  // every emitted resource declares a QI-Core profile
  for (const e of bundle.entry) {
    const r = e.resource as Record<string, unknown>;
    assert.ok(profileOf(r)?.startsWith(QICORE), `${String(r.resourceType)} declares a QI-Core profile`);
  }
});
```
